### Task 2: 實作 exportMarkdown 與 exportJson 函式

**Files:**
- Modify: `history.html`（JS 區塊，`escapeHtml` 函式附近）

找到 JS 區塊中的 `escapeHtml` 函式，在其後新增以下兩個函式：

- [ ] **Step 1: 新增 `exportMarkdown(meeting)` 函式**

```javascript
function exportMarkdown(meeting) {
  const dateStr = (meeting.created_at || meeting.id || '').replace('T', ' ').slice(0, 16);

  const decisions = (meeting.decisions || []);
  const decisionRows = decisions.length
    ? decisions.map(d => {
        const people = (d.related_people || []).join(', ') || '—';
        return `| ${d.id} | ${d.content || ''} | ${d.rationale || '—'} | ${people} | ${d.status || ''} |`;
      }).join('\n')
    : null;

  const actionItems = (meeting.action_items || []);
  const actionRows = actionItems.length
    ? actionItems.map(a =>
        `| ${a.id} | ${a.content || ''} | ${a.assignee || '—'} | ${a.deadline || '—'} | ${a.priority || ''} | ${a.status || ''} |`
      ).join('\n')
    : null;

  const transcript = (meeting.transcript || '').replace(/\n/g, '');

  const lines = [
    `# 會議記錄 ${dateStr}`,
    '',
    `**建立時間：** ${meeting.created_at || ''}`,
    '',
    '---',
    '',
    '## 逐字稿',
    '',
    transcript || '（無）',
    '',
    '---',
    '',
    '## 決議事項',
    '',
  ];

  if (decisionRows) {
    lines.push('| # | 決議內容 | 依據 | 相關人員 | 狀態 |');
    lines.push('|---|---|---|---|---|');
    lines.push(decisionRows);
  } else {
    lines.push('（無）');
  }

  lines.push('', '---', '', '## 待辦事項', '');

  if (actionRows) {
    lines.push('| # | 事項 | 負責人 | 截止日期 | 優先順序 | 狀態 |');
    lines.push('|---|---|---|---|---|---|');
    lines.push(actionRows);
  } else {
    lines.push('（無）');
  }

  const content = lines.join('\n');
  const blob = new Blob([content], { type: 'text/markdown;charset=utf-8' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `meeting-${meeting.id}.md`;
  a.click();
  URL.revokeObjectURL(url);
}
```

- [ ] **Step 2: 新增 `exportJson(meeting)` 函式**

緊接在 `exportMarkdown` 之後新增：

```javascript
function exportJson(meeting) {
  const content = JSON.stringify(meeting, null, 2);
  const blob = new Blob([content], { type: 'application/json;charset=utf-8' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `meeting-${meeting.id}.json`;
  a.click();
  URL.revokeObjectURL(url);
}
```

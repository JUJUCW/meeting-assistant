    if not CATEGORIES_PATH.exists():
        save_categories(_DEFAULT_CATEGORIES)
        return list(_DEFAULT_CATEGORIES)
    try:
        return json.loads(CATEGORIES_PATH.read_text())

        return list(_DEFAULT_CATEGORIES)

    CATEGORIES_PATH.write_text(json.dumps(categories, ensure_ascii=False, indent=2))


def add_category(name: str) -> dict:
    cats = load_categories()
    new_cat = {"id": f"cat-{uuid.uuid4().hex[:8]}", "name": name}
    cats.append(new_cat)
    save_categories(cats)
    return new_cat


def delete_category(cat_id: str) -> bool:
    cats = load_categories()
    new_cats = [c for c in cats if c["id"] != cat_id]
    if len(new_cats) == len(cats):
        return False
    save_categories(new_cats)
    return True
```

    name = body.get("name", "").strip()

@app.delete("/categories/{cat_id}")
def delete_category_endpoint(cat_id: str):
    ok = storage.delete_category(cat_id)
    if not ok:

    result = storage.update_meeting_tags(meeting_id, body)

        raise HTTPException(status_code=404, detail="Meeting not found")

// ── Tag Popover ────────────────────────────────────────────────────────────

let activePopover = null; // { meetingId, selectedCategoryId, selectedTags, el }

function openTagPopover(meetingId, anchorEl) {
  closeTagPopover(false);
  const meeting = allMeetings.find(m => m.id === meetingId);
  if (!meeting) return;

  const selectedCategoryId = meeting.category_id || null;
  const selectedTags = [...(meeting.tags || [])];

  const el = document.createElement('div');
  el.className = 'tag-popover';
  el.id = 'tag-popover';

  function render() {
    const catChips = allCategories.map(c => {
      const isSelected = activePopover.selectedCategoryId === c.id;
      return `<span class="tag-chip category-chip${isSelected ? ' filter-active' : ''}" data-popover-cat="${escapeHtml(c.id)}" style="cursor:pointer;">${escapeHtml(c.name)}</span>`;
    }).join('');

    const tagChips = activePopover.selectedTags.map(t =>
      `<span class="tag-chip filter-active" data-popover-tag="${escapeHtml(t)}" style="cursor:pointer;">${escapeHtml(t)} ✕</span>`
    ).join('');

    el.innerHTML = `
      <div class="tag-popover-section">
        <div class="tag-popover-label">分類</div>
        <div class="tag-popover-chips">${catChips}</div>
      </div>
      <div class="tag-popover-section">
        <div class="tag-popover-label">標籤</div>
        <div class="tag-popover-chips">${tagChips}</div>
        <input class="form-input tag-popover-input" placeholder="輸入新標籤後按 Enter…">
      </div>
    `;

    // Category click
    el.querySelectorAll('[data-popover-cat]').forEach(chip => {
      chip.addEventListener('click', () => {
        const catId = chip.dataset.popoverCat;
        activePopover.selectedCategoryId = activePopover.selectedCategoryId === catId ? null : catId;
        render();
      });
    });

    // Tag remove click
    el.querySelectorAll('[data-popover-tag]').forEach(chip => {
      chip.addEventListener('click', () => {
        const tag = chip.dataset.popoverTag;
        activePopover.selectedTags = activePopover.selectedTags.filter(t => t !== tag);
        render();
      });
    });

    // New tag input
    const input = el.querySelector('.tag-popover-input');
    if (input) {
      input.addEventListener('keydown', e => {
        if (e.key === 'Enter') {
          const val = input.value.trim();
          if (val && !activePopover.selectedTags.includes(val)) {
            activePopover.selectedTags.push(val);
          }
          render();
        }
      });
    }
  }

  activePopover = { meetingId, selectedCategoryId, selectedTags, el };
  render();
  document.body.appendChild(el);

  // Position below anchor
  const rect = anchorEl.getBoundingClientRect();
  const top = rect.bottom + window.scrollY + 6;
  const left = Math.min(rect.left + window.scrollX, window.innerWidth - 296);
  el.style.top = `${top}px`;
  el.style.left = `${left}px`;

  // Close on outside click
  setTimeout(() => {
    document.addEventListener('click', outsideClickHandler);
    document.addEventListener('keydown', escKeyHandler);
  }, 0);
}

function outsideClickHandler(e) {
  if (activePopover && !activePopover.el.contains(e.target)) {
    closeTagPopover(true);
  }
}

function escKeyHandler(e) {
  if (e.key === 'Escape' && activePopover) {
    closeTagPopover(true);
  }
}

async function closeTagPopover(save) {
  if (!activePopover) return;
  document.removeEventListener('click', outsideClickHandler);
  document.removeEventListener('keydown', escKeyHandler);

  if (save) {
    const { meetingId, selectedCategoryId, selectedTags } = activePopover;
    try {
      const r = await fetch(`${API}/meetings/${encodeURIComponent(meetingId)}/tags`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ category_id: selectedCategoryId, tags: selectedTags })
      });
      if (r.ok) {
        const idx = allMeetings.findIndex(m => m.id === meetingId);
        if (idx >= 0) {
          allMeetings[idx].category_id = selectedCategoryId;
          allMeetings[idx].tags = selectedTags;
        }
        renderCurrent();
      }
    } catch (err) {
      console.error('Save tags failed:', err);
    }
  }

  activePopover.el.remove();
  activePopover = null;
}
```

function applyFilters(meetings) {
  if (!activeFilter) return meetings;
  if (activeFilter.type === 'category') {
    return meetings.filter(m => m.category_id === activeFilter.value);
  }
  if (activeFilter.type === 'tag') {
    return meetings.filter(m => (m.tags || []).includes(activeFilter.value));
  }
  return meetings;
}

function renderCurrent() {
  if (lastSearchQuery && lastSearchResults) {
    renderSearchResults(applyFilters(lastSearchResults), lastSearchQuery);
  } else {
    renderList(applyFilters(allMeetings));
  }
  // Highlight active filter chips
  document.querySelectorAll('.tag-chip[data-filter-type]').forEach(chip => {
    const isActive = activeFilter &&
      chip.dataset.filterType === activeFilter.type &&
      chip.dataset.filterValue === activeFilter.value;
    chip.classList.toggle('filter-active', isActive);
  });
}
